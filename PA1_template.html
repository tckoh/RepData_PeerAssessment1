##Function to fill up NA values
match_NA <- function(x, y){

    a <- nrow(x)
    b <- nrow(y)

    for (i in 1:a){
        if(is.na(x[i,1])){
            for (j in 1:b){
                if(x[i,3] == y[j,1]){
                    x[i,1] <- y[j,2]
                    break
                }
            }
        }
    }

    return (x)
}

##Call function match_NA
data_df_No.NA <- match_NA(data_df_No.NA, steps_per_interval_increasing)

wkday_wkend <- function(x){

    a <- nrow(x)    

    for(i in 1:a){
        if (x[i,4] == FALSE){
            x[i,4] <- "weekday";
        }else{
            x[i,4] <- "weekend";
        }
    }

    return(x)
}

data_df_No.NA <- wkday_wkend(data_df_No.NA)
